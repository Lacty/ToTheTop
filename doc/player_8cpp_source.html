184 void Player::teleportingEffect(float sync) {
185  // テレポート中の処理
186  if (isTeleporting_) {
187  elapsedProductionTime_ += sync;
188 
189  // 演出所要時間の半数で演出を区切る
190  if (elapsedProductionTime_ <= (productionTime_ * ofGetFrameRate()) / 2) {
191  /*移動前の座標でプレイヤーが徐々に小さくなる*/
192  size_ -= (p_size_ / productionTime_ / ofGetFrameRate()) * 2;
193  beforePos_ += (p_size_ / productionTime_ / ofGetFrameRate());
194  pos_ = beforePos_;
195  }
196 
197  if (elapsedProductionTime_ > (productionTime_ * ofGetFrameRate()) / 2 &&
198  elapsedProductionTime_ <= (productionTime_ * ofGetFrameRate())) {
199  /*移動先でプレイヤーが徐々に大きくなる*/
200  size_ += (p_size_ / productionTime_ / ofGetFrameRate()) * 2;
201  afterPos_ -= (p_size_ / productionTime_ / ofGetFrameRate());
202  pos_ = afterPos_ + p_size_/2;
203 
204  if (!FindSound(TELEPORT_USE)->isPlaying()) { PlaySound(TELEPORT_USE); }
205  }
206 
207  // 演出所要時間を越えたら終了
208  if (elapsedProductionTime_ > productionTime_ * ofGetFrameRate()) {
209  isTeleporting_ = false;
210  size_ = ofVec2f(p_size_, p_size_);
211  elapsedProductionTime_ = 0;
212  }
213  }
