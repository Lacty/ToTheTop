158 void Player::teleportingEffect(float sync) {
159  // テレポート中の処理
160  if (isTeleporting_) {
161  elapsedProductionTime_ += sync;
162 
163  // 演出所要時間の半数で演出を区切る
164  if (elapsedProductionTime_ <= (productionTime_ * ofGetFrameRate()) / 2) {
165  /*移動前の座標でプレイヤーが徐々に小さくなる*/
166  size_ -= (p_size_ / productionTime_ / ofGetFrameRate()) * 2;
167  beforePos_ += (p_size_ / productionTime_ / ofGetFrameRate());
168  pos_ = beforePos_;
169  }
170 
171  if (elapsedProductionTime_ > (productionTime_ * ofGetFrameRate()) / 2 &&
172  elapsedProductionTime_ <= (productionTime_ * ofGetFrameRate())) {
173  /*移動先でプレイヤーが徐々に大きくなる*/
174  size_ += (p_size_ / productionTime_ / ofGetFrameRate()) * 2;
175  afterPos_ -= (p_size_ / productionTime_ / ofGetFrameRate());
176  pos_ = afterPos_ + p_size_/2;
177  }
178 
179  // 演出所要時間を越えたら終了
180  if (elapsedProductionTime_ > productionTime_ * ofGetFrameRate()) {
181  isTeleporting_ = false;
182  size_ = ofVec2f(p_size_, p_size_);
183  elapsedProductionTime_ = 0;
184  }
185  }
