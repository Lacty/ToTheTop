54 void TeleportState::onCollision(Player* player, Actor* c_actor) {
55  // プレイヤーと衝突判定を行うオブジェクトの必要パラメータを取得
56  auto p_pos = player->getPos();
57  auto p_vel = player->getVel();
58  auto p_size = player->getSize();
59  auto c_pos = c_actor->getPos();
60  //auto c_vel = c_actor->getVel();
61  auto c_size = c_actor->getSize();
62 
63  // Brick以外の物と判定しないように制限
64  if (c_actor->getTag() == BRICK) {
65  // テレポート中は上からの判定を無視する(圧死しない)
66  // Brickに上からぶつかったら加速度を０に(左右への移動量はそのまま)
67  // Brickの上にPlayerの位置を修正
68  if (p_pos.y + p_vel.y < c_pos.y + c_size.y &&
69  (p_pos.y + p_size.y / 3) - p_vel.y > c_pos.y + c_size.y &&
70  p_pos.x + (p_size.x / 10) <= c_pos.x + c_size.x &&
71  p_pos.x + p_size.x - (p_size.x / 10) >= c_pos.x &&
72  p_vel.y < 0) {
73  player->onFloor(true);
74  player->setVel(ofVec2f(0.0f, 0.0f));
75  player->setPos(ofVec2f(p_pos.x, c_pos.y + c_size.y));
76  }
77 
78  // Playerの左辺がBrickの右辺とCollisionした場合
79  else if (p_pos.x < c_pos.x + c_size.x &&
80  p_pos.x + p_size.x > c_pos.x + c_size.x &&
81  p_pos.y - p_vel.y * 2 < c_pos.y + c_size.y &&
82  p_pos.y + p_size.y > c_pos.y) {
83  player->setVel(ofVec2f(0.0f, p_vel.y));
84  // 画面外に押し出されないように調整
85  if (c_pos.x + c_size.x >= ofGetWidth()) {
86  player->setPos(ofVec2f(p_pos.x, c_pos.y + c_size.y));
87  }
88  else {
89  player->setPos(ofVec2f(c_pos.x + c_size.x, p_pos.y));
90  }
91  }
92 
93  // Playerの右辺がBrickの左辺とCollisionした場合
94  else if (p_pos.x + p_size.x > c_pos.x &&
95  p_pos.x < c_pos.x &&
96  p_pos.y - p_vel.y * 2 < c_pos.y + c_size.y &&
97  p_pos.y + p_size.y > c_pos.y) {
98  player->setVel(ofVec2f(0.0f, p_vel.y));
99  // 画面外に押し出されないように調整
100  if (c_pos.x - p_size.x < 0) {
101  player->setPos(ofVec2f(p_pos.x, c_pos.y + c_size.y));
102  }
103  else {
104  player->setPos(ofVec2f(c_pos.x - p_size.x, p_pos.y));
105  }
106  }

126 void TeleportState::movePos(float deltaTime, Player* player, ofxJoystick& input) {
127  auto c_pos = cursor_.lock()->getPos();
128  auto c_vel = cursor_.lock()->getVel();
129  auto c_size = cursor_.lock()->getSize();
130  auto c_speed = cursor_.lock()->getSpeed();
131  auto p_pos = player->getPos();
132  //auto p_vel = player->getVel();
133  auto p_size = player->getSize();
134 
135  // カーソルとプレイヤーの中心座標
136  ofVec2f c_center = ofVec2f(c_pos.x + (c_size.x / 2),
137  c_pos.y + (c_size.y / 2));
138  ofVec2f p_center = p_pos + (p_size / 2);
139 
140  // 左右への移動
141  if (input.isPushing(Input::Left) &&
142  c_pos.x >= 0 &&
143  p_center.distance(c_center - ofVec2f(c_speed, 0.0f)) <= cursor_.lock()->getCircle()) {
144  c_vel.x = -c_speed;
145  cursor_.lock()->setVel(c_vel);
146  }
147  else if (input.isPushing(Input::Right) &&
148  c_pos.x + c_size.x <= ofGetWidth() &&
149  p_center.distance(c_center + ofVec2f(c_speed, 0.0f)) <= cursor_.lock()->getCircle()) {
150  c_vel.x = c_speed;
151  cursor_.lock()->setVel(c_vel);
152  }
153  else {
154  c_vel.x = 0;
155  cursor_.lock()->setVel(c_vel);
156  }
157 
158  // 上下への移動
159  if (input.isPushing(Input::Down) &&
160  p_center.distance(c_center - ofVec2f(0.0f, c_speed)) <= cursor_.lock()->getCircle()) {
161  c_vel.y = -c_speed;
162  cursor_.lock()->setVel(c_vel);
163  }
164  else if (input.isPushing(Input::Up) &&
165  p_center.distance(c_center + ofVec2f(0.0f, c_speed)) <= cursor_.lock()->getCircle()) {
166  c_vel.y = c_speed;
167  cursor_.lock()->setVel(c_vel);
168  }
169  else {
170  c_vel.y = 0;
171  cursor_.lock()->setVel(c_vel);
172  }
173 }
