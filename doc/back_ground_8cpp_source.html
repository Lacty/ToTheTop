104 void BackGround::update(float deltaTime) {
105  deltaTime_ += deltaTime;
106 
107  // 星の処理
108  for (auto& star : stars_) {
109  star->update(deltaTime);
110  }
111 
112  // インターバルが来たら星を生成
113  if (deltaTime_ > interval_) {
114  deltaTime_ = 0;
115  float px = ofRandom(spawnPosMin_.x, spawnPosMax_.x);
116  float py = ofRandom(spawnPosMin_.y, spawnPosMax_.y);
117  float vx = ofRandom(velocityMin_.x, velocityMax_.x);
118  float vy = ofRandom(velocityMin_.y, velocityMax_.y);
119 
120  ofVec3f pos(px, py, 0);
121  ofVec3f vel(vx, vy, 0);
122 
123  stars_.emplace_back(
124  make_unique<Star>(
125  pos,
126  vel,
127  starColor_,
128  ofRandom(extendMin_, extendMax_),
129  0.07,
130  0.13f,
131  ofRandom(widthMin_, widthMax_),
132  ofRandom(heightMin_, heightMax_)
133  )
134  );
135  }
136 
137  // 画面外に出た星を削除する
